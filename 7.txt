#include <stdio.h>

int main()
{
    int n, dec_num;
    int base = 1;
    
    printf("Число в двоичной системе: "); scanf("%i", &n);
    
    int temp = n;
    while(temp)
    { 
        int last_digit = temp % 10; 
        temp = temp / 10; 
        dec_num += last_digit * base; 
        base = base * 2; 

    }
    printf("Число в десятичной системе: %d\n", dec_num);
    return 0;
}
