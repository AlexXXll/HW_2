#include <stdio.h>

int main()
{
    int k, m, h, s;
    
    printf("Введите секунду: ");scanf("%i", &k);
    
    k = k%86400; //В сутках всего 86400 секунд, из-за чего ставлю фильтр на 1-ю секунду новых суток
    h = k / 3600;
    m = (k%3600) / 60;
    s = k%60;
    printf("%d:%d:%d\n", h, m, s);
    return 0;
}
